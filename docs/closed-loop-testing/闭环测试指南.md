```python
#!/usr/bin/env python3
"""
速度限制测试脚本
测试超速指令是否被 safety_control 拒绝
"""

import json
import rclpy
from std_msgs.msg import String

def test_speed_limits():
    rclpy.init()
    node = rclpy.create_node('test_speed_limits')
    
    # 订阅安全决策以监控拒绝
    rejected_count = 0
    def on_decision(msg):
        nonlocal rejected_count
        decision = json.loads(msg.data)
        if decision.get('decision') == 'reject':
            rejected_count += 1
            node.get_logger().warn(f"指令被拒绝: {decision.get('reason')}")
    
    node.create_subscription(String, '/g1/safety/decisions', on_decision, 10)
    
    # 发布运动意图
    pub = node.create_publisher(String, '/voice/cmd/loco', 10)
    
    import time
    time.sleep(2.0)
    
    # 测试超速指令
    over_speed_cmd = {
        "schema_version": "voice_command.v1",
        "source": "test_script",
        "session_id": "test_session",
        "command_id": "test_cmd_001",
        "created_at": time.time(),
        "text": "测试超速",
        "vx": 1.0,  # 超过限制 0.5
        "vy": 0.0,
        "vyaw": 0.0,
        "duration_sec": 1.0
    }
    
    msg = String()
    msg.data = json.dumps(over_speed_cmd, ensure_ascii=False)
    pub.publish(msg)
    
    # 等待处理
    for _ in range(10):
        rclpy.spin_once(node, timeout_sec=0.5)
    
    node.get_logger().info(f"测试完成, 拒绝计数: {rejected_count}")
    
    if rejected_count > 0:
        node.get_logger().info("✅ 速度限制测试通过")
    else:
        node.get_logger().error("❌ 速度限制测试失败")
    
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    test_speed_limits()
```

**预期结果**:

- ✅ 拒绝原因包含速度超限信息

### 4.1 延迟测试

**目标**: 测量端到端系统延迟

**测试脚本**:
```python
#!/usr/bin/env python3
"""
系统延迟测试
测量从ASR输入到运动指令执行的延迟
"""

import json
import rclpy
import time
from std_msgs.msg import String

def test_system_latency():
    rclpy.init()
    node = rclpy.create_node('test_latency')
    
    timestamps = {}
    
    def on_intent(msg):
        timestamps['intent_published'] = time.time()
        node.get_logger().info("运动意图已发布")
    
    def on_safe_cmd(msg):
        timestamps['safe_cmd_published'] = time.time()
        node.get_logger().info("安全指令已发布")
    
    def on_api_request(msg):
        timestamps['api_request'] = time.time()
        node.get_logger().info("API请求已发布")
    
    def on_api_response(msg):
        timestamps['api_response'] = time.time()
        
        # 计算延迟
        if 'asr_input' in timestamps:
            asr_to_intent = (timestamps['intent_published'] - timestamps['asr_input']) * 1000
            intent_to_safe = (timestamps['safe_cmd_published'] - timestamps['intent_published']) * 1000
            safe_to_api = (timestamps['api_request'] - timestamps['safe_cmd_published']) * 1000
            api_to_response = (timestamps['api_response'] - timestamps['api_request']) * 1000
            total_latency = (timestamps['api_response'] - timestamps['asr_input']) * 1000
            
            node.get_logger().info("=== 延迟分析 ===")
            node.get_logger().info(f"ASR → 意图: {asr_to_intent:.1f}ms")
            node.get_logger().info(f"意图 → 安全: {intent_to_safe:.1f}ms")
            node.get_logger().info(f"安全 → API: {safe_to_api:.1f}ms")
            node.get_logger().info(f"API → 响应: {api_to_response:.1f}ms")
            node.get_logger().info(f"总延迟: {total_latency:.1f}ms")
            
            # 验证延迟目标
            if total_latency < 500:  # 500ms目标
                node.get_logger().info("✅ 延迟测试通过")
            else:
                node.get_logger().warn(f"⚠️ 延迟偏高: {total_latency:.1f}ms")
    
    node.create_subscription(String, '/voice/cmd/loco', on_intent, 10)
    node.create_subscription(String, '/g1/safe_cmd/loco', on_safe_cmd, 10)
    node.create_subscription(String, '/api/sport/request', on_api_request, 10)
    node.create_subscription(String, '/api/sport/response', on_api_response, 10)
    
    # 发布ASR输入
    asr_pub = node.create_publisher(String, '/g1/audio/asr', 10)
    
    node.get_logger().info("等待连接...")
    time.sleep(2.0)
    
    # 发送测试输入
    timestamps['asr_input'] = time.time()
    asr_event = {
        "text": "测试延迟",
        "confidence": 0.9,
        "is_final": True,
        "timestamp": timestamps['asr_input']
    }
    
    msg = String()
    msg.data = json.dumps(asr_event, ensure_ascii=False)
    asr_pub.publish(msg)
    
    node.get_logger().info("已发送测试输入，等待响应...")
    
    # 等待完整流程
    for _ in range(20):
        rclpy.spin_once(node, timeout_sec=0.5)
        if 'api_response' in timestamps:
```

```python
#!/usr/bin/env python3
"""
频率限制测试
验证高频指令是否被正确限制
"""

import json
import rclpy
import time
from std_msgs.msg import String

def test_rate_limiting():
    rclpy.init()
    node = rclpy.create_node('test_rate_limit')

    rejected_count = 0

        if decision.get('decision') == 'allow':
            accepted_count += 1
        else:
            rejected_count += 1
            if decision.get('check_details', {}).get('rate_limited'):
                node.get_logger().warn("✅ 频率限制生效")
    
    node.create_subscription(String, '/g1/safety/decisions', on_decision, 10)

            "source": "test_script",
            "session_id": "rate_test",

            "vx": 0.1,
            "vy": 0.0,
            "vyaw": 0.0,
            "duration_sec": 0.5
        }
        
        msg = String()
        msg.data = json.dumps(cmd, ensure_ascii=False)
        intent_pub.publish(msg)
        time.sleep(0.05)  # 50ms间隔，超过5Hz限制
    
    # 等待处理
    time.sleep(2.0)
    
    # 报告结果
    node.get_logger().info("=== 频率限制测试结果 ===")
    node.get_logger().info(f"接受: {accepted_count}")
    node.get_logger().info(f"拒绝: {rejected_count}")
    
    if rejected_count > 0:
        node.get_logger().info("✅ 频率限制测试通过")
    else:
        node.get_logger().error("❌ 频率限制未生效")
    
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
```

## 6. 自动化测试套件

### 6.1 完整测试套件脚本

```python
#!/usr/bin/env python3
"""
完整的自动化测试套件
按顺序执行所有测试并生成报告
"""

import subprocess
import sys
import time
from dataclasses import dataclass
from typing import List

@dataclass
class TestResult:
    name: str
    passed: bool
    duration: float
    message: str

class AutomatedTestSuite:
    def __init__(self):
        self.results: List[TestResult] = []
        self.nodes_running = False
        
    def start_system(self):
        """启动完整的节点系统"""
        print("🚀 启动系统节点...")
        
        # 启动 g1_sim
        self.sim_process = subprocess.Popen(
            ["ros2", "launch", "g1_sim", "g1_sim.launch.py"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE
        )
        time.sleep(3)
        
        # 启动其他节点...
        self.nodes_running = True
        print("✅ 系统节点已启动")
        
    def stop_system(self):
        """停止所有节点"""
        print("🛑 停止系统节点...")
        
        # 停止所有 ROS2 节点
        subprocess.run(["pkill", "-f", "g1_sim"])
        subprocess.run(["pkill", "-f", "g1_interface"])
        subprocess.run(["pkill", "-f", "safety_control"])
        subprocess.run(["pkill", "-f", "voice_bridge"])
        
        self.nodes_running = False
        print("✅ 系统节点已停止")
        
    def run_test(self, test_name: str, test_script: str) -> TestResult:
        """运行单个测试"""
        print(f"\n🧪 运行测试: {test_name}")
        
        start_time = time.time()
        
        try:
            result = subprocess.run(
                ["python3", test_script],
                capture_output=True,
                text=True,
                timeout=30
            )
            
            duration = time.time() - start_time
            passed = result.returncode == 0
            
            message = "测试通过" if passed else f"测试失败: {result.stderr}"
            
            print(f"{'✅' if passed else '❌'} {test_name}: {message}")
            
            return TestResult(
                name=test_name,
                passed=passed,
                duration=duration,
                message=message
            )
            
        except subprocess.TimeoutExpired:
            return TestResult(
                name=test_name,
                passed=False,
                duration=30.0,
                message="测试超时"
            )
        except Exception as e:
            return TestResult(
                name=test_name,
                passed=False,
                duration=0.0,
                message=f"测试异常: {str(e)}"
            )
    
    def generate_report(self):
        """生成测试报告"""
        print("\n" + "="*60)
        print("📊 测试报告")
        print("="*60)
        
        total_tests = len(self.results)
        passed_tests = sum(1 for r in self.results if r.passed)
        pass_rate = (passed_tests / total_tests * 100) if total_tests > 0 else 0
        
        print(f"总测试数: {total_tests}")
        print(f"通过: {passed_tests}")
        print(f"失败: {total_tests - passed_tests}")
        print(f"通过率: {pass_rate:.1f}%")
        print(f"总耗时: {sum(r.duration for r in self.results):.1f}秒")
        
        print("\n详细结果:")
        for result in self.results:
            status = "✅" if result.passed else "❌"
            print(f"{status} {result.name}: {result.message}")
        
        print("="*60)
        
        return pass_rate >= 80  # 80%通过率视为成功
    
    def run_all_tests(self):
        """运行所有测试"""
        try:
            self.start_system()
            time.sleep(2)  # 等待系统稳定
            
            # 定义测试列表
            tests = [
                ("基本运动指令", "test/test_safe_locomotion.py"),
                ("速度限制", "test/test_speed_limits.py"),
                ("ASR输入处理", "test/test_asr_input.py"),
                ("紧急停止", "test/test_emergency_stop.py"),
                ("健康监控", "test/test_health_monitoring.py"),
                ("系统延迟", "test/test_system_latency.py"),
                ("频率限制", "test/test_rate_limiting.py"),
            ]
            
            # 运行所有测试
            for test_name, test_script in tests:
                result = self.run_test(test_name, test_script)
                self.results.append(result)
                time.sleep(1)  # 测试间间隔
            
            success = self.generate_report()
            return 0 if success else 1
            
        finally:
            self.stop_system()

def main():
    suite = AutomatedTestSuite()
    return suite.run_all_tests()

if __name__ == '__main__':
    sys.exit(main())
```
